# 1. Apply the Terraform changes (updates parameter group + task definition)
pixi run apply

# 2. Reboot RDS to apply the static parameter group change
aws rds reboot-db-instance --db-instance-identifier dev-oncokb --region us-east-1
aws rds wait db-instance-available --db-instance-identifier dev-oncokb --region us-east-1

# 3. Re-run init-rds (this ALTER USERs the existing account)
pixi run init-rds

# 4. Restart services
pixi run stop && pixi run start

# 5. Wait ~2 min, then validate
curl -sf http://$(terraform output -raw alb_dns_name)/api/v1/info
